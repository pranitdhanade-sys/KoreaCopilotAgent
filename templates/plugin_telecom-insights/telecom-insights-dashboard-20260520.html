-- 1) 연도별 제조사 고객 수 + YoY
WITH yearly AS (
  SELECT EXTRACT(YEAR FROM Activation_Date) AS yr,
         Manufacturer, COUNT(*) AS cnt
  FROM devices
  WHERE Activation_Date IS NOT NULL
  GROUP BY 1, 2
)
SELECT yr, Manufacturer, cnt,
       LAG(cnt) OVER (PARTITION BY Manufacturer ORDER BY yr) AS prev_cnt,
       ROUND(100.0 * (cnt - LAG(cnt) OVER (PARTITION BY Manufacturer ORDER BY yr))
             / NULLIF(LAG(cnt) OVER (PARTITION BY Manufacturer ORDER BY yr), 0), 1) AS yoy_pct
FROM yearly
ORDER BY Manufacturer, yr;

-- 2) 제조사별 충성도/교체주기
SELECT Manufacturer, COUNT(*) AS customers,
       ROUND(AVG(Brand_Loyalty_Score), 2) AS avg_loyalty,
       ROUND(AVG(Avg_Replacement_Cycle_Months), 1) AS avg_replace_months,
       ROUND(AVG(Months_in_Use), 1) AS avg_months_in_use
FROM devices GROUP BY Manufacturer ORDER BY avg_loyalty DESC;

-- 3) 번들/보조금/보험/Trade-in 조합별 충성도 Top
SELECT Bundling_Status, Subsidy_Type, Device_Insurance, Trade_in_Program,
       COUNT(*) AS cnt,
       ROUND(AVG(Brand_Loyalty_Score), 2) AS avg_loyalty,
       ROUND(AVG(Avg_Data_Usage_GB), 1) AS avg_data_gb
FROM devices GROUP BY 1,2,3,4 HAVING COUNT(*) >= 15
ORDER BY avg_loyalty DESC, cnt DESC LIMIT 20;

-- 4) 제조사 × 번들 매트릭스
SELECT Manufacturer, Bundling_Status, COUNT(*) AS cnt,
       ROUND(AVG(Brand_Loyalty_Score), 2) AS avg_loyalty
FROM devices GROUP BY 1,2 ORDER BY Manufacturer, cnt DESC;
